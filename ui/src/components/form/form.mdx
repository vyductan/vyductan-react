### Basic Usage

Basic Form data control. Includes layout, initial values, validation and submit.

```tsx
import { Button } from "@/components/button";
import { Input } from "@/components/input";
import { Field, Form, useForm } from "@/components/ui/form";
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { z } from "zod";

import { InsertExampleSchema } from "~/api/example/validators";

type Props = {
  initialValues?: {
    username: string;
    password: string;
  };
};
const FormExample = ({ initialValues }: Props) => {
  const queryClient = useQueryClient();

  const insertMutation = useMutation(
    orpc.example.insert.mutationOptions({
      onSuccess: async () => {
        await queryClient.invalidateQueries({
          queryKey: orpc.example.list.key(),
        });
        message.success("Successfully!");
      },
      onError: (error) => {
        apiErrorHandling(error);
      },
    }),
  );

  const form = useForm({
    schema: InsertExampleSchema,
    values: initialValues,
    onSubmit: (values) => {
      insertMutation.mutate(values);
    },
  });

  return (
    <Form form={form}>
      <Field control={form.control} name="username" label="Username">
        <Input />
      </Field>
      <Field control={form.control} name="password" label="Password">
        <Input type="password" />
      </Field>
      <Field control={form.control} name="remember" label="Remember">
        <Checkbox />
      </Field>
      <Field control={form.control} name="role" label="Role">
        <Select options={[{ label: "Admin", value: "admin" }]} />
      </Field>
      <Autocomplete
        allowClear
        options={members.map((member) => ({
          value: member.id,
          label: member.name,
        }))}
      />
      <Field>
        <Button type="submit" loading={insertMutation.isPending}>
          Submit
        </Button>
      </Field>
    </Form>
  );
};
```

### Label Layout (labelCol)

Control label layout using the `labelCol` prop, similar to Ant Design. This is especially useful for horizontal forms where you want to control the width of labels.

```tsx
import { Button } from "@/components/button";
import { Input } from "@/components/input";
import { Field, Form, useForm } from "@/components/ui/form";

const FormExample = () => {
  const form = useForm({
    schema: z.object({
      username: z.string(),
      email: z.string().email(),
    }),
    onSubmit: (values) => {
      console.log(values);
    },
  });

  return (
    <Form form={form} layout="horizontal" labelCol={{ span: 6 }}>
      <Field control={form.control} name="username" label="Username">
        <Input />
      </Field>
      <Field control={form.control} name="email" label="Email">
        <Input />
      </Field>
      <Field>
        <Button type="submit">Submit</Button>
      </Field>
    </Form>
  );
};
```

You can also use responsive column sizing:

```tsx
<Form form={form} layout="horizontal" labelCol={{ xs: 24, sm: 8, md: 6 }}>
  {/* Form fields */}
</Form>
```

### wrapperCol

Control form control wrapper column width and layout.

```tsx
<Form
  form={form}
  layout="horizontal"
  labelCol={{ span: 6 }}
  wrapperCol={{ span: 18 }}
>
  <Field control={form.control} name="username" label="Username">
    <Input />
  </Field>
</Form>
```

### labelAlign

Set label text alignment. Options: `"left"` | `"right"`. Default is `"left"`.

```tsx
<Form form={form} layout="horizontal" labelCol={{ span: 6 }} labelAlign="right">
  <Field control={form.control} name="username" label="Username">
    <Input />
  </Field>
</Form>
```

### labelWrap

Whether label can wrap. Set to `false` to prevent label wrapping (will show ellipsis).

```tsx
<Form form={form} layout="horizontal" labelCol={{ span: 6 }} labelWrap={false}>
  <Field
    control={form.control}
    name="username"
    label="Very Long Username Label"
  >
    <Input />
  </Field>
</Form>
```

### Combined Layout Example

All layout props can be combined together:

```tsx
<Form
  form={form}
  layout="horizontal"
  labelCol={{ span: 6 }}
  wrapperCol={{ span: 16, offset: 2 }}
  labelAlign="right"
  labelWrap={false}
>
  <Field control={form.control} name="username" label="Username">
    <Input />
  </Field>
  <Field control={form.control} name="email" label="Email Address">
    <Input />
  </Field>
  <Field>
    <Button type="submit">Submit</Button>
  </Field>
</Form>
```

### Dynamic Form Item (FieldList)

Add or remove form items dynamically. add function support config initial value.

```tsx
<Form>
  <FieldList control={form.control} name="names">
    {(fields, { add, remove }, { errors }) => (
      <>
        {fields.map((field, index) => (
          <Field
            label={`Passenger ${index + 1}`}
            required={false}
            key={field.key}
          >
            <Field {...field}>
              <Input placeholder="passenger name" style={{ width: "60%" }} />
            </Field>
            {fields.length > 1 ? (
              <Icon
                icon="icon-[lucide--minus-circle]"
                onClick={() => remove(field.name)}
              />
            ) : null}
          </Field>
        ))}
        <Field>
          <Button
            type="dashed"
            onClick={() => add()}
            style={{ width: "60%" }}
            icon={<Icon icon="icon-[lucide--plus]" />}
          >
            Add field
          </Button>
          <Button
            type="dashed"
            onClick={() => {
              add("The head item", 0);
            }}
            style={{ width: "60%", marginTop: "20px" }}
            icon={<Icon icon="icon-[lucide--plus]" />}
          >
            Add field at head
          </Button>
        </Field>
      </>
    )}
  </FieldList>
  <Field>
    <Button type="primary" htmlType="submit">
      Submit
    </Button>
  </Field>
</Form>
```

### Form Modal

```tsx
type Props = {
  open: boolean;
  onOpenChange: (open: boolean) => void;

  initialValues: { id: string; name: string } | undefined;
  onSuccess?: (data: { id: string; name: string }) => void;
};
export const InvoicePeriodFormModal = ({
  open,
  onOpenChange,
  initialValues,
  onSuccess,
}: Props) => {
  const queryClient = useQueryClient();
  const insertMutation = useMutation({
    onSettled: async (data, error) => {
      if (data) {
        message.success("Successfully!");
        await queryClient.invalidateQueries({
          queryKey: "",
        });
        onOpenChange(false);
        onSuccess?.(data.data);
      } else {
        message.error("Failed!");
      }
    },
  });

  const form = useForm({
    schema: InsertSchema,
    values: initialValues,
    onSubmit: (values) => {
      insertMutation.mutate({
        ...values,
      });
    },
  });

  return (
    <>
      <Modal
        title="Form Modal"
        open={open}
        onOpenChange={onOpenChange}
        onOk={async () => {
          await form.submit();
        }}
        okText="Submit"
        confirmLoading={insertMutation.isPending}
      >
        <Form form={form}>
          <div className="grid grid-cols-1 gap-x-6 md:grid-cols-2">
            <Field control={form.control} name="id" label="ID">
              <Input />
            </Field>
            <Field control={form.control} name="name" label="Name">
              <Input />
            </Field>
          </div>
        </Form>
      </Modal>
    </>
  );
};
```
